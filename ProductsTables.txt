create table public."ProductRatings" (
  id uuid not null default gen_random_uuid (),
  product_id uuid not null,
  client_id uuid not null,
  rating smallint not null,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  constraint ProductRatings_pkey primary key (id),
  constraint ProductRatings_product_id_client_id_key unique (product_id, client_id),
  constraint ProductRatings_client_id_fkey foreign KEY (client_id) references auth.users (id) on delete CASCADE,
  constraint ProductRatings_product_id_fkey foreign KEY (product_id) references "Products" (id) on delete CASCADE,
  constraint ProductRatings_rating_check check (
    (
      (rating >= 1)
      and (rating <= 5)
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_productratings_client_id on public."ProductRatings" using btree (client_id) TABLESPACE pg_default;

create table public."Orders" (
  id uuid not null default gen_random_uuid (),
  guest_id uuid null,
  full_name text null,
  phone_number text null,
  address text null,
  governorate_id bigint null,
  shipping_price numeric(10, 2) null,
  status character varying(20) not null default 'pending'::character varying,
  total_amount numeric(10, 2) not null,
  discount numeric(10, 2) null default 0,
  payment_method character varying(20) not null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  promocode_id integer null,
  client_id uuid null,
  constraint orders_pkey primary key (id),
  constraint Orders_client_id_fkey foreign KEY (client_id) references "Clients" (id) on update CASCADE on delete CASCADE,
  constraint Orders_governorate_id_fkey foreign KEY (governorate_id) references "GovernoratesShipping" (id) on update CASCADE,
  constraint fk_orders_promocode foreign KEY (promocode_id) references promo_codes (id) on update CASCADE on delete set null,
  constraint orders_payment_method_check check (
    (
      (payment_method)::text = any (
        (
          array[
            'credit_card'::character varying,
            'paypal'::character varying,
            'cash_on_delivery'::character varying
          ]
        )::text[]
      )
    )
  ),
  constraint orders_status_check check (
    (
      (status)::text = any (
        (
          array[
            'pending'::character varying,
            'paid'::character varying,
            'shipped'::character varying,
            'delivered'::character varying,
            'cancelled'::character varying
          ]
        )::text[]
      )
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_orders_guest_id on public."Orders" using btree (guest_id) TABLESPACE pg_default;

create index IF not exists idx_orders_status on public."Orders" using btree (status) TABLESPACE pg_default;

create index IF not exists idx_orders_governorate_id on public."Orders" using btree (governorate_id) TABLESPACE pg_default;
